def content_tag(name, content="", options={})
  "<#{name}#{tag_options(options) if options}>#{content}</#{name}>"
end

def tag_options(options)
  unless options.empty?
    attributes = []
    options.each do |key, value|
      attributes << %(#{key}="#{value}")
    end
    ' ' + attributes.join(' ')
  end
end
